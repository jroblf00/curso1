002/**
003 * Clase para mostrar por pantalla un mensaje de bienvenida.
004 * 
005 * @author PRG
006 * @version 1.0
007 */
008class Saludo {
009        /**
010         * Metodo que muestra por pantalla un mensaje de bienvenida. * *
011         * 
012         * @param mensaje
013         *            Mensaje a mostrar por pantalla.
014         */
015        void saludar(String mensaje) {
016                System.out.println(mensaje);
017        }
018
019        /**
020         * * Metodo principal de la clase. * *
021         * 
022         * @param args
023         *            Argumentos de la linea de comandos.
024         */
025        public static void main(String args[]) {
026                Saludo objetoSaludo;
027                objetoSaludo = new Saludo();
028                objetoSaludo.saludar("Bienvenido a PRG!");
029        }
030}
